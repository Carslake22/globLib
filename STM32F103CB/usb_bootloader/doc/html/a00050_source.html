157 int usbdfu_control_request(usbd_device *usbd_dev, struct usb_setup_data *req, uint8_t **buf,
158  uint16_t *len, void (**complete)(usbd_device *usbd_dev, struct usb_setup_data *req))
159 {
160  if ((req->bmRequestType & 0x7F) != 0x21)
161  return 0; /* Only accept class request. */
162 
163  switch (req->bRequest) {
164  case DFU_DNLOAD:
165  if ((len == NULL) || (*len == 0)) {
166  usbdfu_state = STATE_DFU_MANIFEST_SYNC;
167  return 1;
168  } else {
169  /* Copy download data for use on GET_STATUS. */
170  prog.blocknum = req->wValue;
171  prog.len = *len;
172  memcpy(prog.buf, *buf, *len);
173  usbdfu_state = STATE_DFU_DNLOAD_SYNC;
174  return 1;
175  }
176  case DFU_CLRSTATUS:
177  /* Clear error and return to dfuIDLE. */
178  if (usbdfu_state == STATE_DFU_ERROR)
179  usbdfu_state = STATE_DFU_IDLE;
180  return 1;
181  case DFU_ABORT:
182  /* Abort returns to dfuIDLE state. */
183  usbdfu_state = STATE_DFU_IDLE;
184  return 1;
185  case DFU_UPLOAD:
186  /* Upload not supported for now. */
187  return 0;
188  case DFU_GETSTATUS: {
189  uint32_t bwPollTimeout = 0; /* 24-bit integer in DFU class spec */
190  (*buf)[0] = usbdfu_getstatus(usbd_dev, &bwPollTimeout);
191  (*buf)[1] = bwPollTimeout & 0xFF;
192  (*buf)[2] = (bwPollTimeout >> 8) & 0xFF;
193  (*buf)[3] = (bwPollTimeout >> 16) & 0xFF;
194  (*buf)[4] = usbdfu_state;
195  (*buf)[5] = 0; /* iString not used here */
196  *len = 6;
197  *complete = usbdfu_getstatus_complete;
198  return 1;
199  }
200  case DFU_GETSTATE:
201  /* Return state with no state transision. */
202  *buf[0] = usbdfu_state;
203  *len = 1;
204  return 1;
205  }
206 
207  return 0;
